Setup Instructions / Tarika

English: Connect Google Sheets by deploying an Apps Script as a Web App.
Urdu: Google Sheet ko connect karne ke liye Apps Script deploy karein.

  1. Open a new Google Sheet.
  2. Go to Extensions → Apps Script.
  3. Delete the default code and paste the code below.
  4. Click Deploy → New Deployment.
  5. Click the Gear icon → Web App.
  6. Set "Execute as" to Me and "Who has access" to Anyone.
  7. Click Deploy, authorize permissions, and copy the Web App URL.
  8. Paste it above, enable sync, and click "Test Connection"!

Apps Script Code

function doPost(e) {
  var result = {};
  try {
    var payload = JSON.parse(e.postData.contents);
    var action = payload.action;
    
    if (action === 'test') {
      result = { status: 'success', message: 'Connected successfully to Google Sheet!' };
    } else if (action === 'syncRow') {
      result = syncRow(payload.row);
    } else if (action === 'deleteRow') {
      result = deleteRow(payload.id, payload.userName);
    } else if (action === 'bulkSync') {
      result = bulkSync(payload.rows);
    } else {
      result = { status: 'error', message: 'Unknown action: ' + action };
    }
  } catch (err) {
    result = { status: 'error', message: err.toString() };
  }
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doOptions(e) {
  return ContentService.createTextOutput("")
    .setMimeType(ContentService.MimeType.TEXT);
}

// Returns the sheet for a given user name. Creates it if it does not exist.
// If no userName is provided, falls back to the default "Daily Work Reports" sheet.
function getOrCreateSheet(userName) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheetName = (userName && userName.toString().trim())
    ? userName.toString().trim()
    : 'Daily Work Reports';
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    var initialHeaders = ['ID', 'Date', 'User Name', 'Company', 'Notes'];
    sheet.getRange(1, 1, 1, initialHeaders.length).setValues([initialHeaders]);
    sheet.getRange(1, 1, 1, initialHeaders.length)
      .setFontWeight('bold')
      .setBackground('#1e3a8a')
      .setFontColor('#ffffff');
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function getHeaders(sheet) {
  var lastCol = sheet.getLastColumn();
  if (lastCol === 0) return [];
  return sheet.getRange(1, 1, 1, lastCol).getValues()[0];
}

function updateHeadersIfNeeded(sheet, currentHeaders, row) {
  var headers = currentHeaders.slice();
  var sheetUpdated = false;
  
  for (var key in row) {
    if (row.hasOwnProperty(key) && key !== 'id') {
      var headerName = getHeaderNameForKey(key);
      if (headers.indexOf(headerName) === -1) {
        headers.push(headerName);
        sheetUpdated = true;
      }
    }
  }
  
  if (sheetUpdated) {
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    sheet.getRange(1, 1, 1, headers.length)
      .setFontWeight('bold')
      .setBackground('#1e3a8a')
      .setFontColor('#ffffff');
  }
  
  return headers;
}

function getHeaderNameForKey(key) {
  var mappings = {
    id: 'ID',
    date: 'Date',
    userName: 'User Name',
    company: 'Company',
    notes: 'Notes'
  };
  if (mappings[key]) return mappings[key];
  return key.charAt(0).toUpperCase() + key.slice(1);
}

function prepareRowValues(headers, row) {
  var values = [];
  var keyMap = {};
  for (var key in row) {
    if (row.hasOwnProperty(key)) {
      var headerName = getHeaderNameForKey(key);
      keyMap[headerName] = row[key];
    }
  }
  for (var i = 0; i < headers.length; i++) {
    var hName = headers[i];
    var val = keyMap[hName];
    values.push((val === undefined || val === null) ? '' : val);
  }
  return values;
}

// Sync a single row. Routes to the sheet named after row.userName.
function syncRow(row) {
  var sheet = getOrCreateSheet(row.userName);
  var headers = getHeaders(sheet);
  
  headers = updateHeadersIfNeeded(sheet, headers, row);
  
  var lastRow = sheet.getLastRow();
  var idColumnValues = lastRow > 1 ? sheet.getRange(2, 1, lastRow - 1, 1).getValues() : [];
  var rowIndex = -1;
  
  for (var i = 0; i < idColumnValues.length; i++) {
    if (idColumnValues[i][0] === row.id) {
      rowIndex = i + 2;
      break;
    }
  }
  
  var rowValues = prepareRowValues(headers, row);
  
  if (rowIndex !== -1) {
    sheet.getRange(rowIndex, 1, 1, rowValues.length).setValues([rowValues]);
    return { status: 'success', message: 'Row updated', index: rowIndex };
  } else {
    sheet.appendRow(rowValues);
    return { status: 'success', message: 'Row appended' };
  }
}

// Delete a row by ID. Searches in the user's own sheet (and fallback sheet).
function deleteRow(id, userName) {
  // Try user sheet first, then fallback sheet
  var sheetsToSearch = [];
  if (userName && userName.toString().trim()) {
    sheetsToSearch.push(getOrCreateSheet(userName));
  }
  var fallback = getOrCreateSheet(null);
  if (sheetsToSearch.indexOf(fallback) === -1) sheetsToSearch.push(fallback);
  
  for (var s = 0; s < sheetsToSearch.length; s++) {
    var sheet = sheetsToSearch[s];
    var lastRow = sheet.getLastRow();
    if (lastRow <= 1) continue;
    var idColumnValues = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
    for (var i = 0; i < idColumnValues.length; i++) {
      if (idColumnValues[i][0] === id) {
        sheet.deleteRow(i + 2);
        return { status: 'success', message: 'Row deleted' };
      }
    }
  }
  return { status: 'error', message: 'Row not found' };
}

// Bulk sync: groups rows by userName and writes each group to its own sheet.
function bulkSync(rows) {
  if (!rows || rows.length === 0) {
    return { status: 'success', message: 'No rows to sync' };
  }
  
  // Group rows by userName
  var groups = {};
  for (var i = 0; i < rows.length; i++) {
    var key = (rows[i].userName && rows[i].userName.toString().trim())
      ? rows[i].userName.toString().trim()
      : 'Daily Work Reports';
    if (!groups[key]) groups[key] = [];
    groups[key].push(rows[i]);
  }
  
  var totalSynced = 0;
  for (var userName in groups) {
    if (!groups.hasOwnProperty(userName)) continue;
    var userRows = groups[userName];
    var sheet = getOrCreateSheet(userName);
    
    // Clear existing data rows
    var lastRow = sheet.getLastRow();
    var lastCol = sheet.getLastColumn() || 5;
    if (lastRow > 1) {
      sheet.getRange(2, 1, lastRow - 1, lastCol).clearContent();
    }
    
    // Update headers and write data
    var headers = getHeaders(sheet);
    for (var j = 0; j < userRows.length; j++) {
      headers = updateHeadersIfNeeded(sheet, headers, userRows[j]);
    }
    var matrix = [];
    for (var j = 0; j < userRows.length; j++) {
      matrix.push(prepareRowValues(headers, userRows[j]));
    }
    sheet.getRange(2, 1, matrix.length, headers.length).setValues(matrix);
    totalSynced += userRows.length;
  }
  
  return { status: 'success', message: 'Bulk sync complete. Synced ' + totalSynced + ' rows across ' + Object.keys(groups).length + ' user sheet(s).' };
}

Daily Report Editor Guide

A comprehensive guide explaining all application options, features, dark mode settings, and system configurations.

1 Team & User Management
  • Add Users: Enter a user name under the Team / Users settings tab and click Add.
  • Rename Users: Click the edit (pen) icon to rename a team member. The system will automatically rename that user across all previously saved records in your history database!
  • Default User: Select a default user to auto-fill the user selector dropdown every time the application is loaded.
2 Dynamic Work Counters
  • Custom Counters: Create new counters under Work Counters by writing a label and choosing one of the 25 custom icons from the icon list.
  • Toggle Switches: Temporarily hide built-in or custom counters from the page layout using the toggle switch. The values are preserved but hidden from active inputs, statistics, and reports.
3 Form Entry & Company Manager
  • Daily Entry: Enter the Date, User, Company, and counter values, then click Submit.
  • Company List: Click the Manage button next to "Select Company" to add, edit, or delete companies from the dropdown menu.
  • Inactive Companies: If you edit a record that uses a deleted company name, the dropdown automatically shows it as (Inactive) so you don't lose data consistency.
4 Duplicate Detection & Auto Merge
  • Smart Alerts: If duplicate entries (same company on the same date) are recorded, a red alert banner lights up. Click the banner to view details in English and Urdu.
  • Auto Merge: Click the Merge Automatically button to combine these duplicate entries: it sums the counter values, merges notes with a divider |, and appends user names together automatically!
5 Dark Slate Mode & Typography
  • Theme Toggling: Go to the Appearance tab to toggle the premium Dark Mode. It transforms the screen into an eye-friendly dark slate layout matching Amazon Central accents.
  • Auto Memory: Theme settings are saved in your browser and automatically loaded on refresh.
  • Custom Select Styles: Dropdowns are custom styled with vector arrows that switch colors dynamically depending on the active theme.
6 Interactive Dashboard & History list
  • Live Metrics: The dashboard cards show overall sums for each active counter. When filters are applied, the dashboard dynamically updates to reflect only the filtered subset!
  • Search and Filter: Use the live search field to instantly filter by user or company, or target specific calendar dates.
  • Edit/Delete: Click the edit (yellow pen) or delete (red trash) icons next to any row in the history list to manage entries.
7 Automatic Backup & Recovery
  • Auto Backups: The system automatically records a daily restore point inside local storage every time you submit a change.
  • Backup Operations: Click the Manage Backups button at the bottom of the page to view date-wise backups. You can restore previous databases, delete unneeded snapshots, or export backups as PDF / JSON files.
8 PDF & High-Res Image Exporter
  • Download PDF: Generates and downloads a clean A4 PDF table including a dedicated sums summary card at the bottom.
  • Download Image: Renders and downloads a premium Daily Report image card. The card automatically prints user summaries, active blockers/notes, and timestamps, perfect for immediate sharing with clients.
9 Local Storage & Data Safety
  • Data Persistence: All entered reports, custom counters, added companies, and user lists are saved directly to browser local storage. Your configuration remains intact even if the tab is closed.
  • Clear Data: Use the Clear All button to reset the database. The system will prompt a confirmation modal to avoid accidental data loss.
10 Import & Restore Backup (JSON)
  • How to Open: Click the Import JSON Backup button at the bottom of the Backup History modal. A dedicated Import window will open.
  • Upload File: Either drag & drop your .json backup file into the upload zone, or click the zone / “click to browse” to select the file from your computer.
  • File Preview: Once a valid file is loaded, a preview card shows the total records, date range, companies in the file, and how many new records will actually be added (exact duplicates are pre-counted and excluded).
  • Merge: Appends only new records from the file to your existing data. Exact duplicates (same date + company + user) are automatically skipped — no manual cleanup needed.
  • Overwrite: Deletes all current records and replaces them with the imported file. A confirmation dialog appears before any data is deleted. Ideal for migrating to a new device.
  • After Import: Data is sorted by date, saved to local storage, an auto-backup is created, and a green success toast appears at the bottom-right of the screen.
  • Compatible Files: Works with any .json file exported via the Download JSON button in the Backup History panel.
